二、排序#

1、冒泡排序#

// 冒泡排序法 JS
for( var i=0; i<a.length-1; i++){
    for( var j=0; j<a.length-1-i; j++){
        // 正序
        if( a[j]>a[j+1]){
            t = a[j+1];
            a[j+1] = a[j];
            a[j] = t;
        }
    }
}
