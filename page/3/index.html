const http = require('http')
// 创建服务
const serve = http.createServer((request, response) => {
    // 设置响应头(文件格式，字符编码)
    response.setHeader("Content-Type","text/plain;charset=utf-8")
    // 返回响应内容
    response.end("hello world")
})
const port = 8888;
const host = "localhost";
// 启动服务并确定域名和端口号
serve.listen(port, host, (err) => {
    if (err) {

    }
    console.log("服务器启动成功")
})

AJAX 即 Asynchronous Javascript And XML，异步 JavaScript 和 XML。是指一种无需重新加载整个网页的情况下，能够更新部分网页的技术。通过Ajax可以在浏览器中向服务器发起异步请求。

XML 称为可扩展标记语言，被设计用来传输和存储数据，和 HTML 类似，不同的是 HTML 日常使用大部分都是预定义标签，XML 全部都是自定义标签，用来表示一些数据。

一、工作原理和基本使用#

1、工作原理#

Ajax 的工作原理相当于在用户和服务器之间加了一个中间层（Ajax引擎），使用户操作与服务器响应异步化。

2、特点#

优点

  • 无需刷新页面来和服务器通信。
  • 允许你根据用户事件来更新部分页面内容。

缺点

  • 没有浏览历史，无法回退。
  • 存在跨域问题。
  • 对 SEO 不友好。

3、XMLHttpRequest 核心对象#

Ajax 的所有操作都是通过 XMLHttpRequest 对象进行的。

4、使用步骤#

（1）创建核心对象#

创建 XMLHttpRequest 核心对象。

const xhr = new XMLHttpRequest();

（2）打开接口，设置数据类型#

通过 open(method, URL) 打开接口。

  • method 请求方法，POST 或者 GET。
  • URL 接口地址。

！这里可以设置数据格式，即请求头，媒体类型有

  • application/json
  • application/x-www-form-urlencoded
  • formdata

xhr.setRequestHeader('Content-Type', 'application/x-www-form-urlencoded')

xhr.open("POST", "http://127.0.0.1/weibo");

（3）发送请求#

发送请求 send(body)

xhr.send( JSON.stringifly( content ));  // 只能传字符串

GET 请求不需要参数， POST 请求需要。

（4）接收数据#

xhr.responseXML 接收XML格式的响应数据。

xhr.responseText 接收文本格式的响应数据。

a. 通过 onreadystatechange 事件接收数据，xhr 对象有 readyState 表示当前的请求状态，状态改变就会触发这个事件。

xhr.onreadystatechange = function (){
	if(xhr.readyState == 4 && xhr.status >= 200 && xhr.status < 300){
		let text = xhr.responseText;
		console.log(text);
	}
}

b. 通过 onload 事件

xhr.onload = function(){
	...xhr.responseText;
}

5、请求和响应状态#

（1）xhr.readyState 当前的请求状态

状态 常量 描述
0 UNSENT 表示 XMLHttpRequest 实例已经生成，但是 open() 方法还没有被调用。
1 OPENED 表示 send()方法还没有被调用，仍然可以使用setRequestHeader()，设定HTTP请求的头信息。
2 HEADERS_RECEIVED 表示 send() 方法已经执行，并且头信息和状态码已经收到。
3 LOADING 表示正在接收服务器传来的body部分的数据，如果responseType属性是text或者空字符串，responseText就会包含已经收到的部分信息。
4 DONE 表示服务器数据已经接收完成，或者本次接收已经失败了。

（2）xhr.status 响应状态，同HTTP协议响应状态码。

一、同源策略#

Same-Origin Policy，同源策略 最早由 Netscape 公司提出，是浏览器的一种安全策略。该策略表示协议、域名、端口号必须完全相同。

违背同源策略就是跨域，主要是针对 AJAX 而言。

二、JSONP 方案#

JSON with Padding，JSONP 是一个非官方的跨域解决方案，只支持get请求，而且需要服务器的配合。

工作原理#

在网页有一些标签天生具有跨域能力，比如 img link iframe script。JSONP就是利用 script 标签的跨域能力来发送请求的。

应用示例#

客户端

<script>
    const script = null;
    // 这是一个事件触发
    ..{
        // 新建一个 script 标签
        script = document.createElement("script");
        // 设置需要跨域的URL
        script.src = "http://.../test?a=1&cb=jp";
        // 插入到页面
        document.body.appendChild(script);
    }
    // 回调函数
    function jp(n){
        // 这边要移除script 标签
        document.body.removeChild(script);
        // 后续操作
    }
</script>

服务器

...
res.end( cb(xxx) );

三、CORS 方案#

Cross-Origin Resource Sharing，CORS，跨域资源共享 是官方的跨域解决方案，它的特点是不需要在客户端做任何特殊的操作，完全在服务器中进行处理，支持 GET 和 POST 请求。是通过设置一个响应头来告诉浏览器，该请求允许跨域，浏览器收到该响应以后就会对响应放行。

// 服务端
app.all("*", (req,res)=>{
    // 设置白名单
    const arrOrigin = [
        "http://xxxxx",
        "xxxx"
    ]
    // 如果访问的URL在白名单中
    if(arrOrigin.oncludes(req.header.origin)){
        // 允许跨域
        res.header("Access-Control-Allow-Origin"),req.header.origin);
        // 允许携带内容
        res.header("Access-Control-Allow-Headers", "content-type")
    }
    next();
})

一、Cookie#

由于HTTP协议是无状态的，而很多服务器业务是必须要用状态。Cookie诞生的最初目的是为了存储网页中的状态信息，以方便服务器端使用。比如判断用户是否是第一次访问网站。

1、流程#

  • 服务器向客户端发送cookie
  • 浏览器保存cookie
  • 之后每次http请求浏览器都会将cookie发送给服务器端

2、服务器处理cookie#

服务器端像客户端发送Cookie是通过HTTP响应报文实现的。

cookie 是一个键值对的存储结构。

  • expires 设置失效时间
  • max-age 有效时间，单位是秒（推荐）
  • httpOnly 仅服务器可修改和可见
// 创建
res.setHeader("Set-Cookie", "user=jack;max-age=3600;httpOnly=true");

// 获取
req.headers.cookie;

// 删除
res.setHeader("Set-Cookie", "user=;max-age=0");

如果不设置时间，则代表是临时会话存储，浏览器关闭后就会消失。

3、客户端处理cookie#

客户端获取 document.cookie

客户端设置 document.cookie = 'name=rose;max-age=3600'

4、缺点#

  • 数量和长度有限制
  • 潜在的安全风险：可能被截取篡改，如果被拦截，就可能会获取到所有的Session信息
  • 用户配置为禁用，有的用户禁用了浏览器或者客户端设备接受cookie的能力，因此限制了这一功能
  • 每次都会随着服务器端发送，具有一定的带宽压力

5、将cookie解析成对象 △#

// 将cookie解析成对象
// ['user=jack', 'name=rose'] --> {user: 'jack', name: 'rose'}
const cookies = req.headers.cookie.split("; ").reduce((p, c) => {
     const [key, value] = c.split("=");
     p[key] = value;
     return p;
}, {});

二、Session#

session 是存储于服务器的对象，是另一种记录客户状态的机制，cookie保存在客户端，session保存在服务器。session运行在服务器，客户端第一次访问服务器时，就可以将客户的登录信息保存。当客户访问其他界面时，可以判断客户的登录状态，做出提示，相当于登录拦截。

1、特点#

  • 存储于服务器上
  • 大小和数量理论上没有限制
  • 传输流量小（只有一个cookie）
  • 安全性较高

2、流程#

  • 客户端：发起登录请求，在请求报文体或者URL中携带登录信息
  • 服务器：接受请求，创建一个session对象，将用户信息保存在session对象中，并创建一个session-id（唯一的）
  • 服务器：返回响应，cookie携带session-id
  • 客户端：接受session-id，保存在本地
  • 客户端：当发起第二次请求的时候，会携带具有session-id的cookie
  • 服务器：通过解析cookie，得到session-id，去session对象中查找当前id是否存在，从而识别用户是不是第一次访问，如果识别用户成功，直接响应通过，否则需要再次查询数据库等操作

四、NodeJS 文件系统 △#

fs 就是 file system 是 NodeJS 核心的文件系统。

const fs = require("fs");

该模块大部分方法提供了两个版本的方法，同步和异步。

  • 同步方法 带sync的方法

    同步方法会阻塞程序的执行

    同步方法通过返回值返回结果

  • 异步方法 不带sync的方法

    异步方法不会阻塞程序的执行

    异步方法都是通过回调函数来返回结果的

1、文件的读取#

（1）简单读取文件#

fs.readFileSync(path[, options])
fs.readFile(path[, options], callback)

  • path 读取文件的路径

  • options 配置对象 默认如下： {encoding:"utf8",flag:"w",mode:0666}

  • callback 回调函数，通过回调函数返回读取到的数据

    err 错误对象

    data 返回的数据（Buffer）

（2）使用 promisify 解决异步问题#

    const fs = require('fs')

    console.log('data',data)

2、文件的写入#

（1）同步写#

打开，写入，==关闭==。

fs.openSync(path[, flags, mode])

  • path 文件路径

  • flag 文件系统标志
    a 追加，如果不存在就创建
    w 用于写，如果不存在就创建，存在覆盖
    r 用于读，如果不存在就异常（默认）

  • mode 设置下一次打开文件时候的权限。
    0o666 110110110 可读可写
    0o111 001001001 可执行
    0o222 010010010 可写
    0o333 100100100 可读

  • 返回文件描述符

fs.writeSync(fd, buffer[, offset, length, position, encoding])

  • fd 文件描述符
  • buffer 要写入的内容
  • offset
  • length
  • position
  • encoding 写人文件的编码，默认utf-8

fs.closeSync(fd)

  • 回调函数err在前是因为NodeJS是==错误优先处理机制==。如果有错误返回错误对象，否则返回null。data 就是文件描述符。
  • 异步得到结果在回调函数中。

3、流式文件操作#

  • fs模块处理文件的缺点：将文件的数据全读到内存中，在把数据写到文件内，会大量占用内存
  • 流（stream）是 Node.js 中处理流式数据的抽象接口，是一组有序的，有起点和终点的字节数据传输手段。可以实现将数据从一个地方流动到另一个地方，其边读取边写入的特点有别于fs模块的文件处理，并且可以做到控制读取文件和写入文件的速度，从而减少内存的占用
  • 流是基于事件的，所有的流对象都用 on(once)绑定事件，并触发
  • 流式文件写入适用于一些比较大的文件，可以分多次向文件中写入内容，有效避免内存溢出的问题。

一、概念#

1、并发、并行#

涉及面试题：并发与并行的区别？

并发（Concurrency）是宏观概念，我分别有任务 A 和任务 B，在一段时间内通过任务间的切换完成了这两个任务，这种情况就可以称之为并发。

并行（Parallelism）是微观概念，假设 CPU 中存在两个核心，那么我就可以同时完成任务 A、B。同时完成多个任务的情况就可以称之为并行。

2、回调函数#

涉及面试题：什么是回调函数？回调函数有什么缺点？如何解决回调地狱问题？

回调函数（Callback）分为同步回调（数组遍历）和异步回调。

使用回调函数容易导致回调地狱问题。回调地狱的根本问题就是：

  1. 嵌套函数存在耦合性，一旦有所改动，就会牵一发而动全身
  2. 嵌套函数一多，就很难处理错误。

当然，回调函数还存在着别的几个缺点，比如不能使用 try catch 捕获错误，不能直接 return。

for ( var i=0; i<3; i++){
    setTimeout(function(){console.log(i);},0);
}
// 输出 3 3 3

for ( var i=0; i<3; i++){
    (function(i){
       setTimeout(function(){console.log(i);},0);
    })(i);
}
// 输出 0 1 2

3、Promise 链#

Promise 实现了链式调用，也就是说每次调用 then 之后返回的都是一个 Promise，并且是一个全新的 Promise，原因也是因为状态不可变。如果你在 then 中 使用了 return，那么 return 的值会被 Promise.resolve() 包装。

4、Promise 方法#

（1）Promise.prototype.then

返回一个Promise 对象。

（2）Promise.prototype.catch

返回一个Promise 对象。

（3）Promise.resolve

返回的Promise对象状态为fulfilled。

（4）Promise.reject

返回一个状态为失败的Promise对象，并将给定的失败信息传递给对应的处理方法。

（5）Promise.all

  • 接收一个数组，数组的每一个元素都的 Promise
  • 返回的是一个 Promise
  • 数组中所有成功，则返回成功，如果有一个失败，则返回失败状态

这个方法返回一个新的promise对象，该promise对象在iterable参数对象里所有的promise对象都成功的时候才会触发成功，一旦有任何一个iterable里面的promise对象失败则立即触发该promise对象的失败。

（6）Promise.race

当iterable参数里的任意一个子promise被成功或失败后，父promise马上也会用子promise的成功返回值或失败详情作为参数调用父promise绑定的相应句柄，并返回该promise对象。

（7）Promise.any

接收一个Promise对象的集合，当其中的一个promise 成功，就返回那个成功的promise的值。

（8）Promise.allSettled

等到所有promises都完成（每个promise返回成功或失败）。
返回一个promise，该promise在所有promise完成后完成。并带有一个对象数组，每个对象对应每个promise的结果。

// 封装 ajax
function ajax(url) {
    return new Promise((resolve, reject)=>{

                    resolve(xhr.responseText);
                } else {
                    reject(xhr.responseText);
                }
            }
        }
        xhr.open("GET", url);
        xhr.send();
    })
}

6、手写 Promise △△#

（1）简单版

const PENDING = "pending";
const RESOLVED = "fulfilled";
const REJECTED = "rejected";

function Promise(exec){
    const that = this;
    that.state = PENDING;
    that.value = null;
    that.resolveedCallbacks = [];
    that.rejectedCallbacks = [];
    
    function resolved(value){
        if( this.state !== PENDING) return;
        that.state = RESOLVED;
        that.value = value;
        that.resolvedCallbacks.map(v=>v(that.value));
    }
    function rejected(value){
        if( this.state !== PENDING) return;
        that.state = REJECTED;
        this.value = value;
        that.rejectedCallbacks.map(v=>v(that.value));
    }
    try{
        exec(resolved, rejected);
    } catch (err){
        rejected(err);
    }
}


Promise.prototype.then = function(onResolved, onRejected){
    const that = this;
    onResolved = typeof onResolved === "function" ? onResolved: v=>v;
    onRejected = typeof onRejected === "function" ? onRejected: err => throw err;
  	switch(that.state){
        case PENDING:
            that.resolvedCallbacks.push(onResolved);
            that.rejectedCallbacks.push(onRejected);
            break;
        case RESOLVED:
            onResolved(that.value);
            break;
        case REJECTED:
            onRejected(that.value);
            break;
    }
}

const _PENDING = "pending";
const _RESOLVED = "fulfilled";
const _REJECTED = "rejected";

function Promise(executor) {
    this.status = _PENDING;
    this.value = undefined;
    this.CallBackList = [];
    function _resolved(value) {
        //   1
        if (this.status !== _PENDING) return;
        
        this.status = _RESOLVED;
        this.value = value;
	    //   2
        if(this.CallBackList.length > 0){
            setTimeout(()=>{
                this.CallBackList.forEach(v => {
                    v.onResolved.call(this);
                });
            })
        }
    };

    function _rejected(reason) {
        if (this.status !== _PENDING) return;
        this.status = _REJECTED;
        this.value = reason;

        if(this.CallBackList.length > 0){
            setTimeout(()=>{
                this.CallBackList.forEach(v => {
                    v.onRejected.call(this);
                });
            })
        }
    }
    try {
        //                  3

Promise.prototype.then = function (onResolved, onRejected) {
    //     4
    return new Promise((resolve, reject) => {
        //   5
        onResolved = typeof onResolved === "function" ? onResolved : value => value;
        onRejected = typeof onRejected === "function" ? onRejected : error => {
            throw error
        };
        
        function _callback(on) {
            //   6
            setTimeout(() => {
                try {
                    let result = on(this.value);
                    // 7
                    if (result instanceof Promise) {
                        result.then(resolve, reject);
                    } else {
                        resolve(result)
                    }
                } catch (err) {
                    reject(err);
                }
            })
        }
        switch (this.status) {
            case _RESOLVED: {
                _callback.call(this, onResolved);
                break;
            }
            case _REJECTED: {
                _callback.call(this, onRejected);
                break;
            }
            //  8
            case _PENDING: {
                CallBackList.push({
                    onResolved() {
                        _callback.call(this, onResolved);
                    },
                    onRejected() {
                        _callback.call(this, onRejected);
                    }
                })
                break;
            }
        }
    })

三、HTTP + TLS = HTTPS#

HTTPS 还是通过了 HTTP 来传输信息，但是信息通过 TLS 协议进行了加密。TLS 协议位于传输层之上，应用层之下。首次进行 TLS 协议传输需要两个 RTT ，接下来可以通过 Session Resumption 减少到一个 RTT。

在 TLS 中使用了两种加密技术，分别为对称加密和非对称加密。

1、两种加密方式#

对称加密

对称加密就是两边拥有相同的秘钥，两边都知道如何将密文加密解密。

这种加密方式固然很好，但是问题就在于如何让双方知道秘钥。因为传输数据都是走的网络，如果将秘钥通过网络的方式传递的话，一旦秘钥被截获就没有加密的意义的。

非对称加密

有公钥私钥之分，公钥所有人都可以知道，可以将数据用公钥加密，但是将数据解密必须使用私钥解密，私钥只有分发公钥的一方才知道。

这种加密方式就可以完美解决对称加密存在的问题。假设现在两端需要使用对称加密，那么在这之前，可以先使用非对称加密交换秘钥。

简单流程如下：首先服务端将公钥公布出去，那么客户端也就知道公钥了。接下来客户端创建一个秘钥，然后通过公钥加密并发送给服务端，服务端接收到密文以后通过私钥解密出正确的秘钥，这时候两端就都知道秘钥是什么了。

2、TLS 握手#

客户端发送一个随机值以及需要的协议和加密方式。

服务端收到客户端的随机值，自己也产生一个随机值，并根据客户端需求的协议和加密方式来使用对应的方式，并且发送自己的证书（如果需要验证客户端证书需要说明）

客户端收到服务端的证书并验证是否有效，验证通过会再生成一个随机值，通过服务端证书的公钥去加密这个随机值并发送给服务端，如果服务端需要验证客户端证书的话会附带证书

服务端收到加密过的随机值并使用私钥解密获得第三个随机值，这时候两端都拥有了三个随机值，可以通过这三个随机值按照之前约定的加密方式生成密钥，接下来的通信就可以通过该密钥来加密解密

通过以上步骤可知，在 TLS 握手阶段，两端使用非对称加密的方式来通信，但是因为非对称加密损耗的性能比对称加密大，所以在正式传输数据时，两端使用对称加密的方式通信。

以上说明的都是 TLS 1.2 协议的握手情况，在 1.3 协议中，首次建立连接只需要一个 RTT，后面恢复连接不需要 RTT 了。

五、WebSocket#

1、概念#

  • 它是一种网络通信协议，是 HTML5 开始提供的一种在单个 TCP 连接上进行全双工通讯的协议。
  • 因为 HTTP 协议有一个缺陷：通信只能由客户端发起
  • 我们都知道轮询的效率低，非常浪费资源（因为必须不停连接，或者 HTTP 连接始终打开）, 因此websocket应运而生。
  • WebSocket用于在Web浏览器和服务器之间进行任意的双向数据传输的一种技术。WebSocket协议基于TCP协议实现，包含初始的握手过程，以及后续的多次数据帧双向传输过程。其目的是在WebSocket应用和WebSocket服务器进行频繁双向通信时，可以使服务器避免打开多个HTTP连接进行工作来节约资源，提高了工作效率和资源利用率。
  • WebSocket目前支持两种统一资源标志符ws和wss，类似于HTTP和HTTPS
  • 浏览器发出webSocket的连线请求，服务器发出响应，这个过程称为握手,握手的过程只需要一次，就可以实现持久连接。

//服务端
const fs = require("fs");
const path = require("path");

// 创建http server
const server = require("http").createServer((req, res) => {
  if (req.url === "/socket.io.js") {
    const filePath = path.resolve(__dirname, "./public/socket.io.js");
    const rs = fs.createReadStream(filePath);
    res.setHeader("Content-Type", "application/javascript;charset=utf8");
    rs.pipe(res);
    return;
  }
  // 响应页面
  const filePath = path.resolve(__dirname, "./public/chat.html");
  const rs = fs.createReadStream(filePath);
  res.setHeader("Content-Type", "text/html;charset=utf8");
  rs.pipe(res);
});

// 创建WebSocket服务
const io = require("socket.io")(server);

// io可以代表所有客户端连接对象
io.on("connect", (socket) => {
  // socket代表当前连接上服务的客户端对象
  socket.on("client_to_server", function (data) {
    console.log(data);
    // 除它以外，通知其他客户端
    socket.broadcast.emit("server_to_client", data);
  });
});

// 启动服务
server.listen(3000, 'localhost');

//客户端
const btn = document.querySelector(".chat-btn");
const content = document.querySelector("textarea");
const input = document.querySelector("#chat-name>input");
const chatContent = document.querySelector(".chat-content");

// 客户端连接上服务端
const socket = io("ws://192.168.10.2:3000");
// socket代表当前连接客户端对象

btn.onclick = function () {
    // 获取当前用户名
    const username = input.value;

    if (!username) {
        alert("请输入用户名！");
        return;
    }

    // 获取用户输入的消息
    const msg = content.value;

    if (!msg) {
        alert("请输入聊天内容！");
        return;
    }

    const time = Date.now();
    // 客户端向服务端发送消息
    socket.emit("client_to_server", {
        username,
        msg,
        time,
    });

    // 显示在你的聊天内容
    const div = document.createElement("div");

    div.innerHTML = `
        <div class="title right-title">${username}  
        ${new Date(time).toLocaleString()}
        </div>
        <div class="right-content">${msg}</div>`;

    chatContent.appendChild(div);

    // 聊天内容清空
    content.value = "";
};

socket.on("server_to_client", function (data) {
    const { username, msg, time } = data;

    // 显示别人的聊天内容

        </div>

1、package 包#

NodeJS 的包基本遵循 CommonJS 规范，包将一组相关的模块组合在一起，形成一套完整的工具，主要文件就是 package.json。

package.json 文件其实就是对项目或者模块包的描述，里面包含许多元信息。比如项目名称，项目版本，项目执行入口文件，项目贡献者等等。npm install命令会根据这个文件下载所有依赖模块。

在项目根目录下执行 npm init，然后根据提示一步步输入相应的内容完成后即可自动创建该文件。

包描述文件包含以下字段：

name - 包名，不能有中文、特殊字符等
version - 包的版本号
description - 包的描述
homepage - 包的官网URL
author - 包的作者
contributors - 包的其他贡献者
description - 项目描述，是一个字符串
keywords - 项目关键字，是一个字符串数组
private - 是否私有，设置为 true 时，npm 拒绝发布
license - 软件授权条款，让用户知道他们的使用权利和限制
bugs - bug 提交地址
repository - 项目仓库地址
homepage - 项目包的官网 URL
dependencies - 生产环境下，项目运行所需依赖，运行包时需要使用的依赖

\> 显示的版本号：
\> 1.12.4 --> 必须是 1.12.4 版本
\> ^1.12.4 --> 必须是 1.12.x 版本，x取最新的
\> ~1.12.4 --> 必须是 1.x.x 版本，x取最新的

devDependencies - 开发环境下，项目所需依赖，构建包时需要使用依赖
scripts - 执行 npm 脚本命令简写，比如 “start”: “react-scripts start”, 执行 npm start 就是运行 “react-scripts start”
bin - 内部命令对应的可执行文件的路径。
eslintConfig - EsLint 检查文件配置，自动读取验证
engines - 项目运行的平台
browseslist - 供浏览器使用的版本列表
files - 被项目包含的文件名数组
