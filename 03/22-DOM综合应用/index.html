一、标签页切换#

方法一

const oH2s = document.querySelectorAll(".h2");
const oLis = document.querySelectorAll(".list");

for (let i = 0; i < oH2s.length; i++) {
    oH2s[i].onclick = function () {
        for (let i = 0; i < oH2s.length; i++) {
            if (oH2s[i] == this) {
                oH2s[i].className = "active";
                oLis[i].className = "show";
            } else {
                oH2s[i].className = "";
                oLis[i].className = "";
            }
        }
    }
}

方法二

const oH2s = document.querySelectorAll(".h2");
const oLis = document.querySelectorAll(".list");

for (let i = 0; i < oH2s.length; i++) {
    oH2s[i].index = i;
    oH2s[i].onclick = function () {
        // 先全部清空
        for (let i = 0; i < oH2s.length; i++) {
            oH2s[i].className = "";
            oLis[i].className = "";
        }
        this.className = "active";
        oLis[this.index].className = "show";

    }
}

方法三（面向对象）

function Tab(btn, content) {
    //把变量或者参数赋值给实例化对象，这样在构造函数和原型对象的方法中都能够访问这个变量
    this.btns = btn;
    this.contents = content;
    //在其他函数中想要使用指向实例化对象的this比较困难，所以用一个that变量保存一个this,方便以后在其他地方使用
    var that = this;
    //对所有的按钮绑定点击事件
    for (var i = 0; i < this.btns.length; i++) {
        //给所有的元素扩展一个下标index属性
        this.btns[i].index = i;
        this.btns[i].onclick = function () {
            //因为所有的Tab切换的核心方法都是一样的，所以我们把方法写在了原型对象上
            //当点击按钮的时候，去执行原型对象上的方法
            //这样所有的Tab切换的执行切换的方法都是同一个，节省效率
            //把当前的this（指向点击的对象）传参传递给方法，方便使用
            that.click(this);
        }
    }
}
Tab.prototype.click = function (that) {
    //使用that来接收参数 参数代表的是点击事件的事件对象
    //先清空所有的内容
    for (var i = 0; i < this.btns.length; i++) {
        this.btns[i].classList.remove("active");
        this.contents[i].classList.remove("active");
    }
    //给当前点击的元素及对应的面板添加类名
    that.classList.add("active");
    this.contents[that.index].classList.add("active");
}
