一、NodeJS HTTP模块#

1、搭建服务器端#

const http = require('http')
// 创建服务
const serve = http.createServer((request, response) => {
    // 设置响应头(文件格式，字符编码)
    response.setHeader("Content-Type","text/plain;charset=utf-8");
    // 返回响应内容
    response.end("hello world");
})
const port = 8888;
const host = "localhost";
// 启动服务并确定域名和端口号
serve.listen(port, host, (err) => {
    if (err) {
        console.log("服务器打开失败" + err);
    }
    console.log("服务器启动成功");
})

2、搭建客户端#

const http = require("http");
const url = "http://localhost:3838";
// 书写请求
const request = http.request(url, res=>{
    // 响应状态码
    console.log(res.statusCode)
    res.on("data“,chunk =>{
 		console.log(chunk.toString());
    }
})
// 发送请求
request.end();

二、Express 框架#

Express 是一个保持最小规模的灵活的 Node.js Web 应用程序开发框架，为 Web 和移动应用程序提供一组强大的功能。

1、安装#

安装模块

npm install express --save

创建一个express项目

express 项目名字

安装 express 的中间件 express-static

npm install express-static --save

安装express 项目生成器

npm install express-generator -g

2、开启服务器#

const express = require("express");
const app = express();
        //port
app.listen(80, function(){
    ...
});

http 默认80端口，https 默认443端口。

3、设置静态资源目录#

app.use( express.static(path.resolve(__dirname, "../xxx")))

！明确到一个文件夹。

！可以指定多个。

4、设置路由#

app.all
app.get( URL, (req, res)=>{})
app.post( URL, (req, res)=>{})

如果指定多个，当遇到访问方法和地址相同时，遇到第一个，默认不会向下执行 – 只会匹配第一个

只能响应一次。

也可以用 next() 继续向下查找是否有满足条件的路由。

第二次响应没有意义。

// 允许所有站点跨域
app.all("*", (req,res)=>{
    res.header("Access-Control-Allow-Origin"),"*");
    next();
})

三、json-server#

用来快速搭建模拟的 REST API 的工具包。

1、开始#

（1）安装

npm i json-server -g

（2）查看版本

json-server -v

（3）准备一个 db.json

// db.json
{
    "sword": [{
        "id": 0,
        "name": "木剑",
        "attr": {
            "atk": 1,
            "parry": 1
        }
    },{
        "id": 1,
        "name": "铜剑",
        "attr": {
            "atk": 5,
            "parry": 3
        }
    },{
        "id": 2,
        "name": "铁剑",
        "attr": {
            "atk": 8,
            "parry": 6
        }
    }{
        "id": 3,
        "name": "钢剑",
        "attr": {
            "atk": 12,
            "parry": 10
        }
    }],
    "shield": [{
        "id": 0,
        "name": "木盾",
        "def": 5
    }]
}

（4）配置服务器

// package.json
// > npm run server
{
    "scripts": {
        "s": "json-server -w db.json -H 127.0.0.1 -p 8888"
    },
    "name": "server",
    "version": "1.0.0",
    "dependencies": {

    }
}
// json-server.json
{
    "port": 80,
    "watch": true,
    "host": "127.0.0.1",
    // 设置静态目录
    "static": "./public",
    "read-only": false,
    // 允许跨域
    "no-cors": false,
    "no-gzip": false,
    "delay": 0
}

（5）开启服务器

json-server --watch db.json
json-server --watch db.json --host 127.0.0.1 --port 8888
npm run server

3、REST API#

REST，Representational State Transfer，表现层状态转移的缩写，它是由罗伊·菲尔丁（Roy Fielding）提出的，是用来描述创建HTTP API的标准方法的，他发现这四种常用的行为（查看（view），创建（create），编辑（edit）和删除（delete））都可以直接映射到HTTP 中已实现的GET,POST,PUT和DELETE方法。

4、操作#

查询 - get

（1）get books

../sword

（2）get id为1

../sword/1
../sword?id=1

（3）单条件查找

../sword?name=铁剑

（4）多条件查找

../sword?name=铁剑&atk=10

（5）全局模糊搜索

../sword?q=剑

（6）分页

../sword?_limit=1&_page=1

（7）排序（asc/desc）

../sword?sort=id&_order=desc
